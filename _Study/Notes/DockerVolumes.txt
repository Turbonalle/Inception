--------------------------------------------------------------------------------
	CREATE VOLUMES WITH DOCKER COMPOSE
--------------------------------------------------------------------------------


version: '3'

services:
	mongodb:
		image: mongo
		ports:
			- 27017:27017
		volumes:
			- db-data:/var/lib/mysql/data
	mongo-express:
		image: mongo-express
		...

volumes:
	db-data:


--------------------------------------------------------------------------------
	DOCKER VOLUME LOCATIONS
--------------------------------------------------------------------------------

WINDOWS		C:\ProgramData\docker\volumes
LINUX		/var/lib/docker/volumes
MAC			/var/lib/docker/volumes

- Inside the volumes folder is a list of all the volumes`
- Each volume has its own unique hash and "/_data" which contains all the files and data that is persisted
	- ex: cfc1ab...ddf22/_data
